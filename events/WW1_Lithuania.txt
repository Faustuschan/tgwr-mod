
add_namespace = ww1_lithuania

# Remove Kingdom of Lithuania
####################
country_event = {
	id = ww1_lithuania.1
	trigger = {
		tag = LIT
		has_country_flag = released_by_germany
		OR = {
			is_subject = no
			AND = {
				is_subject_of = GER
				GER = {
					has_country_flag = great_war_loser
				}
				NOT = {
					has_global_flag = versallies_treaty
				}
			}
		}
	}
	mean_time_to_happen = {
		days = 5
	}
	fire_only_once = yes
	hidden = yes
	immediate = {
		drop_cosmetic_tag = yes
		set_party_name = {
			ideology = neutrality
			long_name = LIT_neutrality_party_long
			name = LIT_neutrality_party
		}
		create_country_leader = {
			name = "Antanas Smetona"
			picture = "gfx/leaders/LIT/portrait_lit_antanas_smetona.dds"
			expire = "1935.1.1"
			ideology = authoritarian_democracy
			traits = {
				trait_authoritarian_democracy
			}
		}
		if = {
			limit = {
				is_subject_of = GER
			}
			GER = {
				end_puppet = LIT
			}
			leave_faction = yes
		}
	}
}

# Death of Mindaugas II
####################
country_event = {
	id = ww1_lithuania.2
	title = ww1_lithuania.2.t
	desc = ww1_lithuania.2.d
	trigger = {
		has_country_leader = {
			name = "Mindaugas II"
			ruling_only = no
		}
		date > 1928.3.24
	}
	mean_time_to_happen = {
		days = 0
	}
	fire_only_once = yes
	option = {
		name = ww1_lithuania.2.a
		kill_ideology_leader = neutrality
		create_country_leader = {
			name = "Vytautas II"
			picture = "gfx/leaders/LIT/LIT_Vytautas_II.dds"
			expire = "1935.1.1"
			ideology = autocracy
			traits = {
				trait_autocracy
			}
		}
	}
}
