	trigger = {
		original_tag = TUR
		has_civil_war = no

			has_country_leader = {
				name = "Ismail Enver Paşa"
				ruling_only = yes
			}
			has_country_leader = {
				name = "Mehmed Talat Paşa"
				ruling_only = yes
			}
			has_country_leader = {
				name = "Mustafa Kemal Paşa"
				ruling_only = yes
			}
			has_country_leader = {
				name = "Mahmud Şevket Paşa"
				ruling_only = yes
			}
		}
		has_country_flag = great_war_loser
		NOT = {
		    has_country_flag = TUR_national_movement_tag
		}
	}
	option = {
		name = generic_ok
		hidden_effect = {
			TUR = {
				country_event = {
					id = ww1_turkey.4
					days = 30
				}
			}
		}
		if = {
			limit = {
				has_country_flag = mustafa_enver_cemal
			}
			set_country_flag = no_mustafa_kemal
		}
		every_country = {
			limit = {
				has_country_flag = great_war_winner
				is_major = yes
			}
			TUR = {
				give_military_access = PREV
			}
		}
		set_politics = {
			ruling_party = neutrality
			elections_allowed = no
		}
		retire_country_leader = yes
		clr_country_flag = talat_enver_cemal
		clr_country_flag = mustafa_enver_cemal
		hidden_effect = {
			country_event = {
				id = ww1_turkey.2
				days = 2
			}
			country_event = {
				id = ww1_turkey.10
				days = 18
			}
		}
		set_country_flag = had_ittihadist_government
		create_country_leader = {
			name = "Ahmet Izzet Paşa"
			picture = "gfx/leaders/TUR/TUR_ahmet_izzet.dds"
			ideology = authoritarian_democracy
			expire = "1935.1.1"
			traits = {
				the_absentee_president
				grand_vizier

			}
		}
	}
}
