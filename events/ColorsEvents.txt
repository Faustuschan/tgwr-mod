		set_cosmetic_tag = SOV_puppet
	}
}
# Remove Soviet-Red Color
####################

	id = color.2

		NOT = { 
		    is_puppet_of = SOV

		}
		has_cosmetic_tag = SOV_puppet
	}

		days = 0
	}

	hidden = yes

		drop_cosmetic_tag = yes
	}
}

		}
	}	
	mean_time_to_happen = {
		days = 0
	}

	hidden = yes

	immediate = {
		drop_cosmetic_tag = yes
	}
}

# Wallachia
####################
country_event = {
	id = color.10
	
	trigger = {
		original_tag = ROM
		MOL = { exists = yes }
		NOT = {
		    any_state = {
				is_core_of = MOL
				NOT = { state = 80 }
                is_owned_by = ROM
            } 
		}
        NOT = {
            AND = {		
                TRA = { exists = no }
				any_state = {
					is_core_of = TRA
					NOT = { state = 80 }
                    is_owned_by = ROM
                }
            }
        }
        NOT = { has_cosmetic_tag = WAL }		
        has_civil_war = no
	}	
	mean_time_to_happen = {
		days = 0
	}

	hidden = yes

	immediate = {
		set_cosmetic_tag = WAL
	}
}
